1. Get the application URL by running these commands:
{{- if .Values.ingress.enabled }}
  https://{{ .Values.ingress.host }}
{{- else }}
  export POD_NAME=$(kubectl get pods --namespace {{ .Values.namespace.name }} -l "app.kubernetes.io/name={{ include "ddoswep-web.name" . }},app.kubernetes.io/instance={{ .Release.Name }}" -o jsonpath="{.items[0].metadata.name}")
  export CONTAINER_PORT=$(kubectl get pod --namespace {{ .Values.namespace.name }} $POD_NAME -o jsonpath="{.spec.containers[0].ports[0].containerPort}")
  echo "Visit http://127.0.0.1:3000 to use your application"
  kubectl --namespace {{ .Values.namespace.name }} port-forward $POD_NAME 3000:$CONTAINER_PORT
{{- end }}

4. View logs:
  kubectl logs -n {{ .Values.namespace.name }} -l app.kubernetes.io/name={{ include "ddoswep-web.name" . }} --tail=100 -f
